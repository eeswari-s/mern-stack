  3.Append  && AppendChild:

  AppendChild:

Examples:
 
   <div id="container">DOM</div>


  <script>
    const div = document.getElementById("container");//<div>dom</div>

  append:
  Example:
  <div id="box"></div>


  <script>
    const box = document.getElementById("box");//<div></div>
    const p1 = document.createElement("p");//<p></p>
    p1.textContent = "Hello World";//<p>Hello World</p>
     const p2 = document.createElement("p");//<p></p>
    p2.textContent = "Welcome to JavaScript";//<p>Welcome to JavaScript</p>
    
    box.append(p1, p2, "Have a great day!");//<div><p>Hello World</p><p>Welcome to JavaScript</p>Have a great day!</div>
  </script>

Example:

<div id="drop" style="width:200px; height:200px; background-color:yellow; margin-top:20px;"></div>

<script>
const drag = document.getElementById('drag');
const drop = document.getElementById('drop');
drop.ondragover= (e)=>e.preventDefault();
drop.ondrop=() =>drop.append(drag);
 
</script>
